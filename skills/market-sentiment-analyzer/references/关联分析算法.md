# 行情-舆情关联分析算法

> 走势原因推导与利好/利空识别算法说明

## 核心逻辑

通过分析行情数据（涨跌幅度、成交量、技术指标）与舆情数据（新闻情绪、关键词）的关联，推导涨跌原因。

---

## 1. 关联分析框架

### 输入数据

```python
market_signals = {
    "price_change": 5.2,           # 涨跌幅%
    "volume_ratio": 1.8,           # 成交量比
    "rsi": 68.5,                   # RSI
    "macd_signal": "golden_cross", # MACD信号
    "trend": "upward"              # 趋势
}

sentiment_signals = {
    "positive_ratio": 0.73,        # 正面舆情占比
    "keywords": ["订单", "突破"],  # 高频关键词
    "news_count": 15               # 新闻数量
}
```

### 关联矩阵

| 行情特征 | 舆情特征 | 关联度 | 解读 |
|----------|----------|--------|------|
| 大涨(>3%) | 正面>70% | 强关联 | 基本面驱动 |
| 大跌(<-3%) | 负面>70% | 强关联 | 利空驱动 |
| 大涨 | 舆情平静 | 弱关联 | 资金抢筹/技术性反弹 |
| 大跌 | 舆情平静 | 弱关联 | 技术性回调 |
| 震荡 | 舆情分化 | 无关联 | 正常波动 |
| 放量上涨 | 订单/业绩 | 强关联 | 业绩超预期 |
| 缩量下跌 | 负面舆情 | 强关联 | 恐慌抛售 |

## 2. 走势原因推导算法

### 算法流程

```
输入：行情数据 + 舆情数据
    ↓
Step 1: 行情特征提取
    - 涨跌幅等级（大涨/小涨/震荡/小跌/大跌）
    - 量能特征（放量/缩量/正常）
    - 技术特征（超买/超卖/金叉/死叉）
    ↓
Step 2: 舆情特征提取
    - 情绪等级（极度乐观/乐观/中性/悲观/极度悲观）
    - 关键词聚类（业绩/订单/政策/技术等）
    - 信息密度（新闻数量）
    ↓
Step 3: 关联匹配
    - 行情-舆情匹配度计算
    - 识别主导因素
    ↓
Step 4: 原因生成
    - 基于规则生成原因摘要
    - 风险提示
```

### 核心代码

```python
class TrendAnalyzer:
    """走势原因分析器"""
    
    # 行情等级阈值
    PRICE_LEVELS = {
        "strong_rise": 5.0,    # 大涨
        "rise": 2.0,           # 小涨
        "fall": -2.0,          # 小跌
        "strong_fall": -5.0    # 大跌
    }
    
    # 情绪等级阈值
    SENTIMENT_LEVELS = {
        "very_positive": 0.8,
        "positive": 0.6,
        "neutral": 0.4,
        "negative": 0.2
    }
    
    def classify_price(self, change_pct):
        """分类涨跌等级"""
        if change_pct >= self.PRICE_LEVELS["strong_rise"]:
            return "strong_rise"
        elif change_pct >= self.PRICE_LEVELS["rise"]:
            return "rise"
        elif change_pct <= self.PRICE_LEVELS["strong_fall"]:
            return "strong_fall"
        elif change_pct <= self.PRICE_LEVELS["fall"]:
            return "fall"
        return "neutral"
    
    def classify_sentiment(self, positive_ratio):
        """分类情绪等级"""
        if positive_ratio >= self.SENTIMENT_LEVELS["very_positive"]:
            return "very_positive"
        elif positive_ratio >= self.SENTIMENT_LEVELS["positive"]:
            return "positive"
        elif positive_ratio <= self.SENTIMENT_LEVELS["negative"]:
            return "negative"
        elif positive_ratio <= self.SENTIMENT_LEVELS["neutral"]:
            return "neutral_sentiment"
        return "neutral"
    
    def analyze_correlation(self, market_data, sentiment_data):
        """
        分析行情与舆情的关联
        
        Returns:
            dict: 关联分析结果
        """
        price_level = self.classify_price(market_data["change_pct"])
        sentiment_level = self.classify_sentiment(sentiment_data["positive_ratio"])
        volume_signal = market_data.get("volume_signal", "normal")
        
        # 关联度计算 (-1 到 1)
        correlation = 0
        
        # 同向加强
        if price_level in ["strong_rise", "rise"] and sentiment_level in ["very_positive", "positive"]:
            correlation = 0.8
        elif price_level in ["strong_fall", "fall"] and sentiment_level in ["negative"]:
            correlation = 0.8
        
        # 反向背离
        elif price_level in ["strong_rise", "rise"] and sentiment_level in ["negative"]:
            correlation = -0.6
        elif price_level in ["strong_fall", "fall"] and sentiment_level in ["very_positive", "positive"]:
            correlation = -0.6
        
        # 量能验证
        if volume_signal in ["high", "extreme_high"]:
            correlation *= 1.2  # 放量加强关联
        
        correlation = max(-1, min(1, correlation))
        
        return {
            "price_level": price_level,
            "sentiment_level": sentiment_level,
            "correlation": round(correlation, 2),
            "volume_signal": volume_signal
        }
```

### 因素分类

```python

    },
    "technology": {
        "positive": ["技术突破", "产品创新", "专利获批", "量产落地"],
        "negative": ["技术落后", "产品替代", "专利纠纷", "量产延期"]

    },

    }
```

### 因素识别算法

```python
def identify_factors(keywords, sentiment_ratio):
    """
    识别利好/利空因素
    
    Args:
        keywords: 舆情关键词列表
        sentiment_ratio: 正面情绪占比
    
    Returns:
        dict: {positive_factors, negative_factors, primary_factor}
    """
    positive_factors = []
    negative_factors = []
    
    for category, words in FACTOR_CATEGORIES.items():
        pos_matches = [k for k in keywords if any(w in k for w in words["positive"])]
        neg_matches = [k for k in keywords if any(w in k for w in words["negative"])]
        
        if pos_matches:
            positive_factors.append({
                "category": category,
                "keywords": pos_matches,
                "weight": len(pos_matches)
            })
        
        if neg_matches:
            negative_factors.append({
                "category": category,
                "keywords": neg_matches,
                "weight": len(neg_matches)
            })
    
    # 排序
    positive_factors.sort(key=lambda x: x["weight"], reverse=True)
    negative_factors.sort(key=lambda x: x["weight"], reverse=True)
    
    # 主导因素
    primary_factor = None
    if sentiment_ratio > 0.6 and positive_factors:
        primary_factor = positive_factors[0]
    elif sentiment_ratio < 0.4 and negative_factors:
        primary_factor = negative_factors[0]
    
    return {
        "positive": positive_factors[:3],  # 最多3个
        "negative": negative_factors[:3],
        "primary": primary_factor
    }
```

### 摘要生成器

```python
def generate_cause_summary(analysis_result, factors, market_data):
    """
    生成走势原因摘要
    
    Args:
        analysis_result: 关联分析结果
        factors: 因素识别结果
        market_data: 行情数据
    
    Returns:
        str: 原因摘要文本
    """
    price_level = analysis_result["price_level"]
    correlation = analysis_result["correlation"]
    change_pct = market_data["change_pct"]
    
    # 选择模板
    if price_level not in CAUSE_TEMPLATES:
        price_level = "neutral"
    
    templates = CAUSE_TEMPLATES[price_level]
    
    # 根据关联度选择
    if abs(correlation) >= 0.6:
        corr_key = "high_correlation"
    else:
        corr_key = "low_correlation"
    
    if corr_key not in templates:
        corr_key = "any" if "any" in templates else list(templates.keys())[0]
    
    template_group = templates[corr_key]
    
    # 根据情绪选择具体模板
    if factors["primary"] and factors["primary"]["category"] in ["fundamental", "policy", "technology"]:
        sentiment_key = "positive" if correlation > 0 else "negative"
    else:
        sentiment_key = list(template_group.keys())[0]
    
    template = template_group.get(sentiment_key, template_group[list(template_group.keys())[0]])
    
    # 填充变量
    primary_factor_name = ""
    if factors["primary"]:
        primary_factor_name = FACTOR_CATEGORIES[factors["primary"]["category"]]["positive"][0] if correlation > 0 else \
                             FACTOR_CATEGORIES[factors["primary"]["category"]]["negative"][0]
    
    summary = template.format(
        primary_factor=primary_factor_name,
        change_pct=change_pct
    )
    
    return summary
```

### 风险检测器

```python
def detect_risks(market_data, sentiment_data):
    """
    检测风险信号
    
    Returns:
        list: 风险列表
    """
    risks = []
    
    for rule in RISK_RULES:
        if rule["condition"](market_data, sentiment_data):
            risks.append({
                "type": rule["name"],
                "level": rule["level"],
                "message": rule["message"]
            })
    
    # 按风险等级排序
    level_order = {"high": 0, "medium": 1, "low": 2}
    risks.sort(key=lambda x: level_order.get(x["level"], 3))
    
    return risks
```

## 6. 完整分析流程

```python
def full_analysis(industry_name, market_data, sentiment_data):
    """
    完整的走势原因分析
    
    Returns:
        dict: 完整分析结果
    """
    analyzer = TrendAnalyzer()
    
    # 1. 关联分析
    correlation = analyzer.analyze_correlation(market_data, sentiment_data)
    
    # 2. 因素识别
    factors = identify_factors(
        sentiment_data.get("keywords", []),
        sentiment_data.get("positive_ratio", 0.5)
    )
    
    # 3. 生成摘要
    summary = generate_cause_summary(correlation, factors, market_data)
    
    # 4. 风险检测
    risks = detect_risks(market_data, sentiment_data)
    
    return {
        "industry": industry_name,
        "analysis_time": time.strftime("%Y-%m-%d %H:%M:%S"),
        "correlation": correlation,
        "factors": factors,
        "cause_summary": summary,
        "risks": risks,
        "overall_assessment": "看涨" if correlation["correlation"] > 0.5 else "看跌" if correlation["correlation"] < -0.5 else "震荡"
    }
```
