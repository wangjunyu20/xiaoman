```python
import requests
import json

class CaiLianSheClient:
    """财联社客户端"""
    
    BASE_URL = "https://www.cls.cn/v3"
    HEADERS = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Referer": "https://www.cls.cn/"
    }
    
    def __init__(self):
        self.session = requests.Session()
        self.session.headers.update(self.HEADERS)
    
    def search_news(self, keyword, limit=20):
        """搜索新闻"""
        url = f"{self.BASE_URL}/search"
        params = {
            "keyword": keyword,
            "limit": limit
        }
        
        try:
            resp = self.session.get(url, params=params, timeout=10)
            data = resp.json()
            
            results = []
            for item in data.get("data", {}).get("items", []):
                results.append({
                    "title": item.get("title", ""),
                    "content": item.get("content", ""),
                    "time": item.get("time", ""),
                    "source": "财联社",
                    "url": f"https://www.cls.cn/detail/{item.get('id', '')}"
                })
            return results
        except Exception as e:
            print(f"财联社请求失败: {e}")
            return []
    
    def fetch_telegrams(self, limit=50):
        """获取电报快讯"""
        url = f"{self.BASE_URL}/telegraphs"
        params = {"limit": limit}
        
        try:
            resp = self.session.get(url, params=params, timeout=10)
            data = resp.json()
            
            results = []
            for item in data.get("data", []):
                results.append({
                    "content": item.get("content", ""),
                    "time": item.get("time", ""),
                    "source": "财联社电报",
                    "url": f"https://www.cls.cn/telegraph"
                })
            return results
        except Exception as e:
            print(f"财联社电报获取失败: {e}")
            return []
```

```python
# 搜索行业新闻
def search_industry_news(industry_name, days=7):
    """搜索行业相关新闻"""
    from kimi_search import kimi_search
    
    # 构建搜索查询
    queries = [
        f"{industry_name} 最新动态 2026",
        f"{industry_name} 政策 利好",
        f"{industry_name} 订单 业绩",
        f"{industry_name} 龙头 行情"
    ]
    
    all_results = []
    for query in queries:
        results = kimi_search(query, limit=5)
        all_results.extend(results)
    
    # 去重并格式化
    seen_urls = set()
    formatted = []
    for item in all_results:
        if item['url'] not in seen_urls:
            seen_urls.add(item['url'])
            formatted.append({
                "title": item['title'],
                "content": item.get('snippet', ''),
                "url": item['url'],
                "source": "互联网搜索",
                "time": ""
            })
    
    return formatted[:20]  # 返回前20条
```

### 情绪词典

### 情绪评分算法

```python
def analyze_sentiment(text, title=""):
    """
    分析文本情绪倾向
    
    Args:
        text: 正文内容
        title: 标题（权重更高）
    
    Returns:
        dict: {sentiment, score, keywords}
    """
    combined = title * 2 + text  # 标题权重加倍
    
    pos_count = sum(1 for word in POSITIVE_WORDS if word in combined)
    neg_count = sum(1 for word in NEGATIVE_WORDS if word in combined)
    
    total = pos_count + neg_count
    if total == 0:
        return {"sentiment": "neutral", "score": 0.5, "keywords": []}
    
    score = pos_count / total
    keywords = [w for w in POSITIVE_WORDS if w in combined] + \
               [w for w in NEGATIVE_WORDS if w in combined]
    
    if score > 0.6:
        sentiment = "positive"
    elif score < 0.4:
        sentiment = "negative"
    else:
        sentiment = "neutral"
    
    return {
        "sentiment": sentiment,
        "score": round(score, 2),
        "keywords": list(set(keywords))[:5]  # 最多5个关键词
    }
```

## 5. 数据整合

### 舆情数据聚合器

```python
class SentimentAggregator:
    """舆情数据聚合器"""
    
    def __init__(self):
        self.cls_client = CaiLianSheClient()
        self.guba_client = EastMoneyGubaClient()
    
    def collect_sentiment(self, industry_name, leader_codes, days=3):
        """
        聚合多源舆情数据
        
        Args:
            industry_name: 行业名称
            leader_codes: 龙头股代码列表
            days: 获取近几天数据
        
        Returns:
            dict: 聚合后的舆情数据
        """
        all_news = []
        
        # 1. 财联社新闻
        cls_news = self.cls_client.search_news(industry_name, limit=20)
        all_news.extend(cls_news)
        
        # 2. 东财股吧
        for code in leader_codes[:3]:  # 取前3只龙头股
            posts = self.guba_client.get_stock_posts(code, pages=2)
            all_news.extend(posts)
        
        # 3. 互联网搜索
        search_results = search_industry_news(industry_name, days)
        all_news.extend(search_results)
        
        # 情绪分析
        analyzed = []
        pos_count = 0
        neg_count = 0
        all_keywords = []
        
        for item in all_news:
            sentiment = analyze_sentiment(item.get("content", ""), item.get("title", ""))
            item.update(sentiment)
            analyzed.append(item)
            
            if sentiment["sentiment"] == "positive":
                pos_count += 1
            elif sentiment["sentiment"] == "negative":
                neg_count += 1
            
            all_keywords.extend(sentiment["keywords"])
        
        total = pos_count + neg_count
        positive_ratio = pos_count / total if total > 0 else 0.5
        
        # 关键词统计
        from collections import Counter
        top_keywords = [word for word, _ in Counter(all_keywords).most_common(10)]
        
        return {
            "total_count": len(analyzed),
            "positive_count": pos_count,
            "negative_count": neg_count,
            "positive_ratio": round(positive_ratio, 2),
            "sentiment_level": "high" if positive_ratio > 0.7 else "low" if positive_ratio < 0.3 else "neutral",
            "keywords": top_keywords,
            "news": analyzed[:30]  # 返回前30条
        }
```

### 缓存实现

```python
import json
import os
import time
from pathlib import Path

class SentimentCache:
    """舆情数据缓存"""
    
    CACHE_DIR = "/workspace_investment/output/market_analysis/cache"
    
    def __init__(self):
        Path(self.CACHE_DIR).mkdir(parents=True, exist_ok=True)
    
    def get_cache_path(self, source, key):
        """获取缓存文件路径"""
        date_str = time.strftime("%Y%m%d")
        return f"{self.CACHE_DIR}/{source}_{key}_{date_str}.json"
    
    def get(self, source, key, max_age=300):
        """获取缓存数据"""
        cache_path = self.get_cache_path(source, key)
        
        if not os.path.exists(cache_path):
            return None
        
        # 检查过期
        mtime = os.path.getmtime(cache_path)
        if time.time() - mtime > max_age:
            return None
        
        with open(cache_path, 'r', encoding='utf-8') as f:
            return json.load(f)
    
    def set(self, source, key, data):
        """设置缓存数据"""
        cache_path = self.get_cache_path(source, key)
        with open(cache_path, 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
```

## 注意事项

1. **频率限制**: 财联社/东财有反爬机制，请求间隔建议>10秒
2. **数据准确性**: 股吧数据噪音大，需要清洗过滤
3. **情绪偏差**: 情绪分析基于词典，可能有误差
4. **时效性**: 新闻情绪具有时效性，建议当天使用
5. **版权合规**: 抓取数据仅供研究，勿商用传播
